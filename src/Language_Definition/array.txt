arr[num] points = [5, 10, 12, 4, 7, 2];

points[2] -= 6; #[5,10,6,4,7,2]
points[4] = 10; #[5,10,6,4,10,2]
points.add(17); #[5,10,6,4,10,2,17]

#addPoints function use add(expression) method to add new element to the array
func void addPoints(num newPoint){
    points.add(newPoint);
    result("Add points");
}

    if(i > points[i]){
        addPoints(points[i] + i);
