#User readInput() keyword to read user input from console
dict info = {
    "Username": "",
    "Password": "",
    "Email": ""
};

str userInput = "";
result("Welcome to Simplify");

f_loop(num i = 0 to info.size()) exist {
    arr[str] keys = ["Username", "Password", "Email"];
    str curKey = keys[i];
    result("Please enter your info: ");
    userInput = readInput("");
    info[curKey] = userInput;
}
